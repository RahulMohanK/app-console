Create Procedure addData
@EmpId NVARCHAR(50),
@Name NVARCHAR(50),
@Phone NVARCHAR(50),
@Email NVARCHAR(50),
@Dob DATETIME,
@StaffType int,
@Subject NVARCHAR(50)
AS
BEGIN
    INSERT into Staffs(EmpId,Name,Phone,Email,Dob,StaffType,CreatedDate,UpdatedDate) values (@EmpId,@Name,@Phone,@Email,@Dob,@StaffType,GETDATE(),GETDATE())
    INSERT into AdministrativeStaff(StaffId,Designation) values((Select Id from Staffs where Name = @Name),@Subject)
END
